class DRMStation(SimpleStation):
    """A specialization of the SimpleStation Receiver class, with added functionality
    for processing of DRM motions.

    :param x: xyz location of the station.
    :type x: numpy array (3,)
    :param internal: mark if its an internal or external DRM station
    :type internal: bool
    :param filter_results: (optional) whether a filter should be applied to station response
    :type filter_results: bool
    :param filter_params: (optional) filter parameter settings (see :class:`Receivers.SimpleStation`)
    :type filter_params: dict
    :param name: (optional) station name 
    :type name: str

    .. note::

        This class is intended for internal use by :class:`Receivers.DRMBox`

    """

    def __init__(self, x, drmbox, internal, filter_results=False, filter_parameters={}, name=""):

        SimpleStation.__init__(self, x, filter_results=filter_results, filter_parameters=filter_parameters, name=name)
        self.drmbox = drmbox
        self.internal = internal

    def add_to_response(self, z, e, n, t):
        """See :class:`Receivers.SimpleStation.add_to_response`

        """

        self.drmbox.__report_time_vector__(t)
        return SimpleStation.add_to_response(self, z, e, n, t)

    def is_internal(self):
        """Check whether current station is in the internal boundary of the DRM box. 

        The __init__ method may be documented in either the class level
        docstring, or as a docstring on the __init__ method itself.

        Either form is acceptable, but the two should not be mixed. Choose one
        convention to document the __init__ method and be consistent with it.

        Note:
            Do not include the `self` parameter in the ``Args`` section.

        Returns:
            bool. `True` if this station is internal to the DRM box. 

        """
        return self.internal

class Plane:
    """A helper class to define the planes of the sides of the DRM box. 

    :param v0: Origin ofset in xyz coordinates. 
    :type v0: numpy array (3,)
    :param v1: Direction vector of 1 local plane coordinate in xyz coordinates. (Magnitude is the length of the side).
    :type v1: numpy array (3,)
    :param v2: Direction vector of 2 local plane coordinate in xyz coordinates. (Magnitude is the length of the side).
    :type v2: numpy array (3,)
    :param k: Plane tag
    :type k: int
    :param internal: True if plane is represents an internal boundary of the DRM box. 
    :type internal: bool
    :param xi1: Natural coordinate grid of the plane. Defines spacing of stations in the 1 direction. Should be monotonically increasing spanning the interval [0,1] (including the edges)
    :type xi1: numpy array (N1,)
    :param xi2: Natural coordinate grid of the plane. Defines spacing of stations in the 2 direction. Should be monotonically increasing spanning the interval [0,1] (including the edges)
    :type xi2: numpy array (N2,)

    Note:

    This class is intended for internal use by :class:`Receivers.DRMBox`.

    """

    def __init__(self, v0, v1, v2, k, internal, xi1, xi2):

        self.v0 = v0
        self.v1 = v1
        self.v2 = v2
        self.k = k
        self.internal = internal
        self.stations = np.empty((len(xi1), len(xi2)), dtype=np.int32)
        self.xi1 = xi1
        self.xi2 = xi2


    def set_station_id(self,i,j,station_id):
        """Set the station_id for the station at the (i,j) in the local plane of the grid.

        """
        self.stations[i,j] = station_id

    def write_me_into_dataset(self, f):
        """Writes the information of the plane into an HDF5 dataset.

        Args:
            :param f: HDF5 file or group handle. 
            :type f: h5py object.

        """
        f.create_dataset("v0", data=self.v0)
        f.create_dataset("v1", data=self.v1)
        f.create_dataset("v2", data=self.v2)
        f.create_dataset("internal", data=self.internal)
        f.create_dataset("stations", data=self.stations)
        f.create_dataset("xi1", data=self.xi1)
        f.create_dataset("xi2", data=self.xi2)

    """A class to generate receiver layout useful in DRM.

    .. figure::  ../docs/images/drmbox.png
       :scale: 60%
       :align:   center

    
    :param pos:  Center point of the DRM box in xyz coordinates. 
    :type pos: (numpy array (3,))
    :param Nelem:  Number of elements (stations) in each direction. ``Nelem = [Nx, Ny, Nz]``
    :type Nelem: (int)
    :param h:  Spacings in each direction ``h = [hx, hy, hz]``
    :type h: (double)
    :param azimuth:  Azimuthal orientation of the box. 
    :type azimuth: (double)
    :param crust:  Crust properties. 
    :type crust: (:obj:`fkdrm.CrustModel`)
    :param filter_results:  (See :obj:`fkdrm.Receivers.SimpleStation`)
    :type filter_results: (bool)
    :param filter_parameters: (See :obj:`fkdrm.Receivers.SimpleStation`)
    :type filter_parameters: (dict)
    :param name: 
    :type name: (str)
    :param x_qa:  List of points where QA data is requested. 
    :type x_qa: list of numpy array (3,)

    .. note::
        Side lengths of the DRM box are ``[Nx*hx, Ny*hy, Nz*hz]`` up to the interior
        boundary of the box. Exterior boundary has side lengths: ``[(Nx+2)*hx, (Ny+2)*hy, (Nz+1)*hz]``

    """
    

    def __init__(self, pos, Nelem, h, azimuth=0.,crust=0, filter_results=False, filter_parameters={}, name="", x_qa = []):

        #Data members
        self.x0 = np.array(pos)
        self.h = np.array(h)
        self.Nelem = np.array(Nelem)
        self.azimuth = azimuth
        self.filter_results = filter_results
        self.filter_parameters = filter_parameters
        self.name = name
        self.nstations = 0
        self.nplanes = 0
        self.planes = []
        self.rcvlist = []
        self.x_qa = []
        self.tstart = np.infty
        self.tend = -np.infty
        self.dt = 0
        self.xmax = -np.infty
        self.xmin =  np.infty
        self.ymax = -np.infty
        self.ymin =  np.infty
        self.zmax = -np.infty
        self.zmin =  np.infty

        #Define QA stations
        self.x_qa.append(self.x0)
        for x in x_qa:
            self.x_qa.append(x)

        self.__create_QA_stations__()
        self.__create_DRM_stations__()
        self.__initialize_HDF5_dataset__()

    def __create_QA_stations__(self):
        """__create_QA_stations__

        Private method in DRMBox. User should not need to call. 

        """
        for x in self.x_qa:
            self.__newstation__([x[0],x[1],0],0,0,-1,False)


    def __create_DRM_stations__(self):
        """__create_DRM_stations__

        Private method in DRMBox. User should not need to call. 
        
        """

        #DRM box orientation (TODO: add azimuthal rotation)
        e1 = np.array([1.,0.,0.])
        e2 = np.array([0.,1.,0.])
        e3 = np.array([0.,0.,1.])

        #Inner boundary
        internal = True
        Nx = self.Nelem[0]
        Ny = self.Nelem[1]
        Nz = self.Nelem[2]
        hx = self.h[0]
        hy = self.h[1]
        hz = self.h[2]
        lx = Nx * hx
        ly = Ny * hy
        lz = Nz * hz
        xi_x = np.linspace(0., 1., Nx+1)
        xi_y = np.linspace(0., 1., Ny+1)
        xi_z = np.linspace(0., 1., Nz+1)

        v0 = self.x0 + (lx/2)*e1 - (ly/2)*e2
        v1, v2 = lz*e3, ly*e2
        xi1, xi2 = xi_z, xi_y
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 + (lx/2)*e1 + (ly/2)*e2
        v1, v2 = lz*e3, -lx*e1
        xi1, xi2 = xi_z, xi_x[1:-1]
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 - (lx/2)*e1 + (ly/2)*e2
        v1, v2 = lz*e3, -ly*e2
        xi1, xi2 = xi_z, xi_y
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 - (lx/2)*e1 - (ly/2)*e2
        v1, v2 = lz*e3, lx*e1
        xi1, xi2 = xi_z, xi_x[1:-1]
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 - (lx/2)*e1 - (ly/2)*e2 + lz*e3
        v1, v2 = lx*e1, ly*e2
        xi1, xi2 = xi_x[1:-1], xi_y[1:-1]
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)


        #Outer boundary
        internal = False
        Nx = self.Nelem[0]+2
        Ny = self.Nelem[1]+2
        Nz = self.Nelem[2]+1
        hx = self.h[0]
        hy = self.h[1]
        hz = self.h[2]
        lx = Nx * hx
        ly = Ny * hy
        lz = Nz * hz
        xi_x = np.linspace(0., 1., Nx+1)
        xi_y = np.linspace(0., 1., Ny+1)
        xi_z = np.linspace(0., 1., Nz+1)
        

        v0 = self.x0 + (lx/2)*e1 - (ly/2)*e2
        v1, v2 = lz*e3, ly*e2
        xi1, xi2 = xi_z, xi_y
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 + (lx/2)*e1 + (ly/2)*e2
        v1, v2 = lz*e3, -lx*e1
        xi1, xi2 = xi_z, xi_x[1:-1]
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 - (lx/2)*e1 + (ly/2)*e2
        v1, v2 = lz*e3, -ly*e2
        xi1, xi2 = xi_z, xi_y
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 - (lx/2)*e1 - (ly/2)*e2
        v1, v2 = lz*e3, lx*e1
        xi1, xi2 = xi_z, xi_x[1:-1]
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        v0 = self.x0 - (lx/2)*e1 - (ly/2)*e2 + lz*e3
        v1, v2 = lx*e1, ly*e2
        xi1, xi2 = xi_x[1:-1], xi_y[1:-1]
        self.__new_DRM_plane__(v0,v1,v2,xi1,xi2,internal)

        StationList.__init__(self, self.rcvlist)


    def __newstation__(self,p,i,j,k,internal):
        """__newstation__

        Private method in DRMBox. User should not need to call. 
        
        """

        x, y, z = p[0], p[1], p[2]
        station_id = self.nstations
        newrcv = DRMStation([x,y,z], self, internal, filter_results=self.filter_results, filter_parameters=self.filter_parameters, name=self.name+".{}.{}.{}".format(i,j,k))
        self.rcvlist.append(newrcv)
        self.xmax = max(x, self.xmax)
        self.xmin = min(x, self.xmin)
        self.ymax = max(y, self.ymax)
        self.ymin = min(y, self.ymin)
        self.zmax = max(z, self.zmax)
        self.zmin = min(z, self.zmin)
        self.nstations += 1
        return station_id


    def __new_DRM_plane__(self,v0,v1,v2,xi,eta,internal):
        """__new_DRM_plane__

        Private method in DRMBox. User should not need to call. 
        
        """
        newplane = Plane(v0,v1,v2,self.nplanes,internal,xi,eta)
        self.planes.append(newplane)
        for i, j in product(np.arange(xi.size), np.arange(eta.size)):
            xi_, eta_ = xi[i], eta[j]
            p = v0 + xi_*v1 + eta_*v2
            station_id = self.__newstation__(p,i,j,self.nplanes,internal) - len(self.x_qa)
            newplane.set_station_id(i,j,station_id)
        self.nplanes += 1


    def __report_time_vector__(self, t):
        """__report_time_vector__

        Private method in DRMBox. User should not need to call. 
        
        """
        self.tstart = min(self.tstart,t[0])
        self.tend = max(self.tend,t[-1])
        self.dt = max(self.dt, t[1] - t[0])


    def __initialize_HDF5_dataset__(self):
        """__initialize_HDF5_dataset__

        Private method in DRMBox. User should not need to call. 
        
        """
        if rank > 0:
            return

        #Form filename and create HDF5 dataset
        if self.name == "":
            self.name = "DRMcase"
        fname = self.name + ".h5drm"
        self.h5file = h5py.File(fname, mode="w")

        #Create groups
        self.grp_DRM_Data = self.h5file.create_group("/DRM_Data")
        self.grp_DRM_QA_Data = self.h5file.create_group("/DRM_QA_Data")
        self.grp_DRM_Planes = self.h5file.create_group("/DRM_Planes")
        self.grp_DRM_Metadata = self.h5file.create_group("/DRM_Metadata")
        self.grp_DRM_ProgramMetadata = self.grp_DRM_Metadata.create_group("ProgramMetadata")

        print "\n\nFKDRM - DRMBox - Creating dataset at: ", fname
        
        #Write planes info
        self.grp_DRM_Planes.create_dataset("nplanes", data=self.nplanes)

        for i,plane in enumerate(self.planes):
            grp_this_plane = self.grp_DRM_Planes.create_group("plane_{0:02.0f}".format(i))
            plane.write_me_into_dataset(grp_this_plane)

        #Write DRM minimum Metadata
        self.grp_DRM_Metadata.create_dataset("fmax", data=0.0)
        self.grp_DRM_Metadata.create_dataset("tstart", data=0.0)
        self.grp_DRM_Metadata.create_dataset("tend", data=0.0)
        self.grp_DRM_Metadata.create_dataset("dt", data=0.0)
        self.grp_DRM_Metadata.create_dataset("program_used", data="fkdrm")
        self.grp_DRM_Metadata.create_dataset("download_url", data="")
        self.grp_DRM_Metadata.create_dataset("reference_doi", data="")
        self.grp_DRM_Metadata.create_dataset("created_by", data="")
        self.grp_DRM_Metadata.create_dataset("created_on", data="")
        self.grp_DRM_Metadata.create_dataset("drmbox_xmax", data=self.xmax)
        self.grp_DRM_Metadata.create_dataset("drmbox_xmin", data=self.xmin)
        self.grp_DRM_Metadata.create_dataset("drmbox_ymax", data=self.ymax)
        self.grp_DRM_Metadata.create_dataset("drmbox_ymin", data=self.ymin)
        self.grp_DRM_Metadata.create_dataset("drmbox_zmax", data=self.zmax)
        self.grp_DRM_Metadata.create_dataset("drmbox_zmin", data=self.zmin)
        self.grp_DRM_Metadata.create_dataset("drmbox_x0", data=self.x0)
        self.grp_DRM_Metadata.create_dataset("h", data=self.h)

        #Write FKDRM Metadata
        self.grp_DRM_ProgramMetadata.create_dataset("Nelem", data=self.Nelem)
        self.grp_DRM_ProgramMetadata.create_dataset("azimuth", data=self.azimuth)
        self.grp_DRM_ProgramMetadata.create_dataset("nstations", data=self.nstations)

    def __write_HDF5_DRM_output__(self):
        """__write_HDF5_DRM_output__

        Private method in DRMBox. User should not need to call. 
        
        """
        
        if rank > 0:
            return

        self.grp_DRM_Metadata["dt"][()] = self.dt
        self.grp_DRM_Metadata["tstart"][()]  = self.tstart
        self.grp_DRM_Metadata["tend"][()]  = self.tend

        
        #Compute global time vector
        t_drm = np.arange(self.tstart, self.tend, self.dt)
        nt = len(t_drm)

        n_qa_stations = len(self.x_qa)
        n_recording_stations = self.nstations - n_qa_stations

        data_location = np.arange(0,n_recording_stations, dtype=np.int32)*3

        #Create data and QA datasets
        velocity = self.grp_DRM_Data.create_dataset("velocity", (3*n_recording_stations, nt), dtype=np.double, chunks=(3,nt))
        xyz = self.grp_DRM_Data.create_dataset("xyz", (n_recording_stations, 3), dtype=np.double)
        internal = self.grp_DRM_Data.create_dataset("internal", [n_recording_stations], dtype=np.bool)
        id_data_location = self.grp_DRM_Data.create_dataset("data_location", data=data_location)
        velocity_qa = self.grp_DRM_QA_Data.create_dataset("velocity", (3*n_qa_stations, nt), dtype=np.double, chunks=(3,nt))
        xyz_qa = self.grp_DRM_QA_Data.create_dataset("xyz", (n_qa_stations, 3), dtype=np.double)


        #Write velocity data into data each dataset
        for i, rcv in enumerate(self):
            x = rcv.get_pos()
            z,e,n,t = rcv.get_response()
            zz = interpolator(t,z,t_drm)
            ee = interpolator(t,e,t_drm)
            nn = interpolator(t,n,t_drm)
            if i < n_qa_stations:
                velocity_qa[3*i,:] = ee
                velocity_qa[3*i+1,:] = nn
                velocity_qa[3*i+2,:] = zz
                xyz_qa[i,:] = x
            else:
                I = i - n_qa_stations
                velocity[3*I,:] = ee
                velocity[3*I+1,:] = nn
                velocity[3*I+2,:] = zz
                xyz[I,:] = x
                internal[I] = rcv.is_internal()

        self.h5file.close()
        print "\n\nFKDRM - DRMBox - Done\n\n "
