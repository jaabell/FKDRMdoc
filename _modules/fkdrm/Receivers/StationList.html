[docs]class StationList(Receiver):
    """This is a list of SimpleStations.

    If you wan't to have more than one point in space where you need
    to record motions, use this class. 

    :param stations: A list of SimpleStations
    :type stations: ``list`` containing :obj:`SimpleStation`

    Example::

        sta1 = SimpleStation([20,20,0])
        sta2 = SimpleStation([20,40,0])

        stations = StationList([sta1, sta2])

    """


    def __init__(self, stations):

        self.rcvlist = []
        for station in stations:
            self.rcvlist.append(station)
        self.icurrent = 0

        """ Iterate the list of stations:

        Example::

            #Obtain a StaitionList iterator
            station_iterator = iter(stations)

        Example::

            #Iterate over all stations using a for loop
            for sta in stations:
                print sta

        """
        self.icurrent = 0
        for rcv in self.rcvlist:
            rcv.__iter__()
        return self

    

[docs]    def next(self): 
        """ Get the next element in the station list.

        Example::

            #Iterate over all stations
            iter = iter(stations)
            next_station = next(iter)

        """
        if self.icurrent == len(self.rcvlist):
            self.icurrent = 0
            raise StopIteration
        else:
            self.current = self.rcvlist[self.icurrent]
            self.icurrent += 1
            return self.current.next()
