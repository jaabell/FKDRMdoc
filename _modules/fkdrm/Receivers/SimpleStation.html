[docs]class SimpleStation(Receiver):
    """This simple receiver stores response in memory. 

    Internally, numpy arrays are used for storage. Optional parameters allow filtering
    of the response before outputting, althoiugh it is always stored
    raw (unfiltered), therefore the user can experiment with
    different filtering settings.

    Results can be saved through the :func:`save` method and loaded with the :func:`load` function.
    for posterior use. 

    :param x: xyz location of the station.
    :type x: numpy array (3,)
    :param filter_results: (optional) whether a filter should be applied to station response
    :type filter_results: bool
    :param filter_parameters: (optional) filter parameter settings (see below)
    :type filter_parameters: dict
    :param name: (optional) station name, used for plotting and other actions.
    :type name: str

    Filtering is performed using the :mod:`scipy.signal` module. A Butterworth filter 
    (:obj:`scipy.signal.butter`) is designed and applied using the acausal two-way
    filter (:obj:`scipy.signal.filtfilt`) with constant padding using the edge values 
    of the response as padding values. 

    Filter parameter settings:

    * ``filter_parameters['fmax']=10``: `(double)` Cut-off frequency for the low-pass filter in Hz. 
    * ``filter_parameters['btype']='low``: `(str)` Filter type 
    * ``filter_parameters['order']=3``: `(int)` Order of the filter. 

    Example::

        #Create a station at (10,10,0), setting filtering 
        # to an order 3 filter with a 3Hz cutoff.
        station = SimpleStation([10,10,0], 
            filter_results=True, 
            filter_parameters={'fmax':5., 'order':3})

    """
    #Default filter parameters
    filter_params = {}
    filter_params["fmax"] = 10.
    filter_params["btype"] = 'low'
    filter_params["order"] = 4


    def __init__(self, x, filter_results=False, filter_parameters={}, name=""):
        #Constructor
        self.x = x
        self.returned = False
        self.initialized = False
        self.filter = filter_results
        self.name = name
        if self.filter:
            for key in self.filter_params:
                if key in filter_parameters:
                    self.filter_params[key] = filter_parameters[key]

    def dofilter(self, x, dt, Npad=-1):
        
        fmax = self.filter_params["fmax"]
        order = self.filter_params["order"]
        btype = self.filter_params["btype"]
        fnyq = 1/(2*dt)
        Wcrit = fmax/fnyq
        B, A = sig.butter(4, Wcrit, btype=btype)
        Nt = len(x)
        if Npad == -1:
            Npad=len(x)
        y = sp.pad(x, (Npad,Npad), "constant", constant_values=(x[0],x[-1])) 
        y = sig.filtfilt(B, A, y)
        return y[Npad:(Npad+Nt)]

    

[docs]    def add_to_response(self, z, e, n, t):
        if not self.initialized:
            # print "SimpleStation.py - adding to response (init)"
            self.z = z
            self.e = e
            self.n = n
            self.t = t
            self.dt = t[1] - t[0]
            self.tmin = t.min()
            self.tmax = t.max()
            self.initialized = True
        else:
            # print "SimpleStation.py - adding to response (add)"
            dt = t[1] - t[0]
            tmin = min(self.tmin, t.min())
            tmax = max(self.tmax, t.max())
            if dt != self.dt:
                print "Warning dt = {} != self.dt = {}. Resampling at lowest rate".format(dt, self.dt)
                dt = max(dt, self.dt)
            tnew = sp.arange(tmin, tmax, dt)
            zz = interpolator(self.t, self.z, tnew)
            zz += interpolator(t, z, tnew)
            ee = interpolator(self.t, self.e, tnew)
            ee += interpolator(t, e, tnew)
            nn = interpolator(self.t, self.n, tnew)
            nn += interpolator(t, n, tnew)
            self.z = zz
            self.e = ee
            self.n = nn
            self.t = tnew

        # print "max zen = ({}, {}, {}) dt = {}  nt = {}".format(max(self.z),max(self.e),max(self.n),self.dt, len(self.t))

[docs]    def get_response(self, do_filter=True):        
        """Return the recorded response of the station. 
        
        :param do_filter: Will/won't filter if filter parameters have been set. (Most useful to disable filtering before return)
        :type do_filter: bool
    
        :returns: Z (down), E (east), N (north), t (time) response of the station. 
        :retval: tuple containing numpt arrays with z, e, n, t reponse (shape (Nt,))

        Example::

            z,e,n,t = station.get_response()

        """
        if self.filter and do_filter:
            dt = self.t[1] - self.t[0]
            z = self.dofilter(self.z, dt)
            e = self.dofilter(self.e, dt)
            n = self.dofilter(self.n, dt)
        else:
            z, e, n = self.z, self.e, self.n
        # return z[self.t>=0], e[self.t>=0], n[self.t>=0], self.t[self.t>=0]
        return z, e, n, self.t



[docs]    def get_response_integral(self, ntimes=1, do_filter=True):
        """Return the time-integral of the recorded response of the station. 
        
        :param ntimes: How many times to time-integrate.
        :type ntimes: int
        :param do_filter: Will/won't filter if filter parameters have been set. (Most useful to disable filtering before return)
        :param do_filter: bool
    
        :returns: Z (down), E (east), N (north), t (time) response of the station integrated ``ntimes`` in time
        :retval: tuple containing numpt arrays with z, e, n, t reponse (shape (Nt,))

        Example::

            z,e,n,t = station.get_response_integral()

        """
        if ntimes == 0:
            return self.get_response(do_filter=do_filter)
        elif ntimes > 0:
            z,e,n,t = self.get_response_integral(ntimes=ntimes-1)
        else:
            print "Station.get_response_integral() - ntimes must be >= 0. Got ntimes = {}".format(nint)
        z_integrated = cumtrapz(z, t, initial=0.)
        e_integrated = cumtrapz(e, t, initial=0.)
        n_integrated = cumtrapz(n, t, initial=0.)
        return z_integrated, e_integrated, n_integrated, t

        """Return the time-derivative of the recorded response of the station. 
        
        :param ntimes: How many times to differentiate in time.
        :type ntimes: int
        :param do_filter: Will/won't filter if filter parameters have been set. (Most useful to disable filtering before return)
        :param do_filter: bool
    
        :returns: Z (down), E (east), N (north), t (time) response of the station differentiated ``ntimes`` in time
        :retval: tuple containing numpt arrays with z, e, n, t reponse (shape (Nt,))

        Example::

            z,e,n,t = station.get_response_derivative()

        """
        if ntimes == 0:
            return self.get_response(do_filter=do_filter)
        elif ntimes > 0:
            z,e,n,t = self.get_response_derivative(ntimes=ntimes-1)
        else:
            print "Station.get_response_derivative() - ntimes must be >= 0. Got ntimes = {}".format(nint)
        z_differentiated = sp.gradient(z, t)
        e_differentiated = sp.gradient(e, t)
        n_differentiated = sp.gradient(n, t)
        return z_differentiated, e_differentiated, n_differentiated, t

        """ Save the current station and all its data into a binary file (uses cPickle).

        :param fname: File name to save to. 
        :type fname: str 

        Example::

            station.save("station.pickle")

        """
        fid = open(fname,"wb")
        cPickle.dump(self, fid)
        fid.close()
