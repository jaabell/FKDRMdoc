[docs]    def __init__(self, crust, source, receiver):
        """Constructor

        Arguments:
        ==========
        :param crust: Crustal model used by the simulation. 
        :type crust: :class:`CrustModel`
        :param source: Source model(s). Choose from :class:`fkdrm.Sources`
        :type source: :class:`fkdrmBase.SourceModel`
        :param receiver: Receiver station(s). Choose from :class:`fkdrm.Receivers`
        :type receiver: :class:`fkdrmBase.Receiver`

        """
        assert isinstance(crust, CrustModel), "crust must be an instance of the fkdrmBase.CrustModel class"
        assert isinstance(source, SourceModel), "crust must be an instance of the fkdrmBase.SourceModel class"
        assert isinstance(receiver, Receiver), "crust must be an instance of the fkdrmBase.Receiver class"

        self.crust = crust 
        self.source = source 
        self.receiver= receiver
        self.rank = rank
        self.nprocs = nprocs

            # print "setting {} = {} (formerly = {})".format(key, value, getattr(self, key, value))
            setattr(self, key, value)

        """Run the simulation. 
        
        Arguments:
        ==========
        :param debug_subgreen: Print internal debug information.
        :type debug_subgreen: bool
        :param debug_mpi: Print debug information concerning parallel runs .
        :type debug_mpi: bool
        :param progressbar: Show fancy progressbar (uses tqdm)
        :type progressbar: bool

        """

        if self.rank != 0:
            progressbar = False

        nrcv = self.receiver.get_nreceivers()
        nsrc = self.source.get_nsources()
        npairs = nrcv * nsrc
        if self.rank == 0:
            print "FKDRM Running "
            print " Number of sources   = ", nsrc
            print " Number of receivers = ", nrcv
            print " Total src-rcv pairs = ", npairs
            print " dt   = ", self.dt
            print " nfft = ", self.nfft
            print " "
            print " Using {} processes. ".format( self.nprocs)
            print " "
        if progressbar and found_tqdm and (self.rank == 0):
            pbar = tqdm(total=npairs, unit="pairs")
            pbar.set_description("FKDRM Progress")    

        if nprocs == 1 or rank == 0:
            next_pair = rank
            skip_pairs = 1
        else :
            next_pair = rank-1
            skip_pairs = nprocs-1

        ipair = 0
        request_list = []
        data_list = []
        for  i_source, source in enumerate(self.source):
            for i_receiver, receiver in enumerate(self.receiver):
                if ipair == next_pair:
                    x_src, angles, stf, tt = source.get_data()

                    if debug_mpi:
                        print debugstring.format(rank=rank, nprocs=nprocs, ipair=ipair, next_pair=next_pair, skip_pairs=skip_pairs)
                    if nprocs == 1 or (rank > 0 and nprocs > 1):
                        x_rcv = receiver.get_pos()
                        thiscrust = copy.deepcopy(self.crust)

                        pf, df, lf = angles[0], angles[1], angles[2]
                        sx, sy, sz = x_src[0], x_src[1], x_src[2]
                        rx, ry, rz = x_rcv[0], x_rcv[1], x_rcv[2]
                        if sz < rz:
                            print "frkdrm.run() - Error source (at {}) should be below receiver (at {})".format(sz, rz)
                        if debug_subgreen:
                            print "Crust Before:"
                            thiscrust.printme()
                        src = thiscrust.split_at_depth(sz)
                        rcv = thiscrust.split_at_depth(rz)
                        if debug_subgreen:
                            print "Crust After:"
                            thiscrust.printme()
                            print "x_src = {} x_rcv = {}".format(x_src, x_rcv)
                            print "  src = {}   rcv = {}".format(src, rcv)
                            print ""
                            print "Running fkdrm"
                        
                        sigma=self.sigma
                        nfft=self.nfft
                        dt=self.dt
                        taper=self.taper
                        tb=self.tb
                        smth=self.smth
                        wc1=self.wc1
                        wc2=self.wc2
                        pmin=self.pmin
                        pmax=self.pmax
                        dk=self.dk
                        kc=self.kc
                        nx=self.nx
                        sigma=self.sigma
                        smth=self.smth
                        wc1=self.wc1
                        wc2=self.wc2
                        pmin=self.pmin
                        pmax=self.pmax
                        dk=self.dk
                        kc=self.kc
                        d,a,b,rho,qa,qb = thiscrust.get_model_params()              
                        mb = thiscrust.get_nlayers()
                        stype,updn = 2,0  #Source type double-couple, compute up and down going wave
                        x = np.sqrt((sx-rx)**2 + (sy - ry)**2)
                        if debug_subgreen:
                            print "    mb = ", mb
                            print "    src = ", src
                            print "    rcv = ", rcv
                            print "    stype = ", stype
                            print "    updn = ", updn
                            print "    d = ", d
                            print "    a = ", a
                            print "    b = ", b

                            print "    qa = ", qa
                            print "    qb = ", qb

                        if debug_mpi:
                            print "P{} working on ipair = {} @ ({}, {}, {}) src @ ({}, {}, {})".format(rank, ipair, x_rcv[0], x_rcv[1], x_rcv[2], x_src[0], x_src[1], x_src[2])

                        #Execute the core subgreen fortran routing                      
                        tdata,z,e,n,t0 = core.subgreen(mb,src,rcv,stype,updn,d,a,b,rho,qa,qb,dt,nfft,tb,nx,sigma,smth,wc1,wc2,pmin,pmax,dk,kc,taper,x,pf,df,lf,sx,sy,rx,ry)
                        nt = len(z)
                        t = np.arange(0,nt*dt,dt)+tt+t0

                        #Resample stf to current dt
                        stf.set_dt(dt)
                        dstf, tstf = stf.get_data()

                        #Convolve with STF
                        if len(dstf) == 1:
                            z_stf = z*dstf[0]
                            e_stf = e*dstf[0]
                            n_stf = n*dstf[0]
                        else:
                            tstf_resampled = np.arange(0, tstf.max(), dt)
                            dstf_resampled = interp1d(tstf, dstf)(tstf_resampled)
                            z_stf = sig.convolve(z,dstf_resampled,mode="same")
                            e_stf = sig.convolve(e,dstf_resampled,mode="same")
                            n_stf = sig.convolve(n,dstf_resampled,mode="same")

                        if rank > 0:
                            # new_request_list = []

                                print "    P{} sending results to P0.".format(rank)
                            ant = np.array([nt], dtype=np.int32).copy()
                            # req1 = comm.Isend([data_list[-1], 1, MPI.INT], dest=0, tag=2*ipair)
                            comm.Send(ant, dest=0, tag=2*ipair)

                            # request_list.append(req1)

                            data = np.empty((nt,4), dtype=np.float64)
                            data[:,0] = z_stf
                            data[:,1] = e_stf
                            data[:,2] = n_stf
                            data[:,3] = t
                            # req2 = commostm.Isend([data_list[-1],nt,MPI.DOUBLE], dest=0, tag=2*ipair+1)
                            comm.Send(data, dest=0, tag=2*ipair+1)

                            # request_list.append(req2)

                            next_pair += skip_pairs

                    #Sum at receiver
                    if rank == 0:

                        if nprocs > 1:
                            skip_pairs_remotes = nprocs-1
                            remote = ipair % skip_pairs_remotes + 1
                            if debug_mpi:
                                print "P0 waiting for {} to send results for ipair = {}.".format(remote, ipair)
                            ant = np.empty(1, dtype=np.int32)
                            comm.Recv(ant, source=remote, tag=2*ipair)
                            nt = ant[0]
                            data = np.empty((nt,4), dtype=np.float64)
                            comm.Recv(data, source=remote, tag=2*ipair+1)
                            z_stf = data[:,0]
                            e_stf = data[:,1]
                            n_stf = data[:,2]
                            t = data[:,3]
                            if debug_mpi:
                                print "P0 got results from {} for ipair = {}.".format(remote, ipair)
                        next_pair += 1

                        receiver.add_to_response(z_stf,e_stf,n_stf,t)


                        if progressbar and found_tqdm:
                            pbar.update(1)
                        else:
                            print "pair, rcv, src = ", ipair,i_receiver,i_source
                    #End if
                else:
                    # print "ipair = {} != next_pair = {}".format(ipair, next_pair)
                    pass
                ipair += 1

        if rank == 0:
            self.receiver.write_response()
            if progressbar and found_tqdm:
                pbar.close()
